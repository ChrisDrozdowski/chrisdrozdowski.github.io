112  std::vector<double> generate_data(long rows, long offset = 0, long count = -1) {
113  if (0 == rows || offset >= rows || 0 == count)
114  return std::vector<double>();
115  //const double x0 = si.get_x0();
116  //const double dx = si.get_dx();
117  if (sampling_interval_default == x0_ || sampling_interval_default == dx_)
118  return std::vector<double>();
119  // If overflow, it will be inf or -inf, which is okay for values.
120  if (offset < 0) offset = 0;
121  if (count < 0) count = rows;
122  // Adjust count so that it doesn't go past number of rows.
123  if (offset + count > rows) count = rows - offset;
124  std::vector<double> data(count);
125  double partial_sum = x0_;
126  // First advance to before offset.
127  // Must use addition in a loop, not multiplication
128  // which leads to cumulative error.
129  for (int i = 0; i < offset; ++i) partial_sum += dx_;
130  // Now generate.
131  for (long i = 0; i < count; ++i) {
132  data[i] = partial_sum;
133  partial_sum += dx_;
134  }
135  return data;
136  }

138  private:
139  // Not able to get long name (OrgLab limitation).
140  bool parse(std::wstring wstr) {
141  reset();
142  if (0 == wstr.length())
143  return false;
144  const std::size_t max_len = 512;
145  // Prevent buffer overflow.
146  if (wstr.length() > max_len)
147  wstr.resize(max_len);
148  /*
149  Note: Long name isn't in the string!
150 
151  Sampling interval string will be in 3 forms:
152  #1: x0 = <number>\r\ndx = <number>
153  #2: x0 = <number>\r\ndx = <number> (<units>)
154  #3: <number> (<units>) the number is dx, x0 is assumed to be 0.
155 
156  But, for our purposes we want it in either of these forms:
157  x0 = <number>\r\ndx=<number>
158  x0 = <number>\r\ndx=<number> (<units>)
159 
160  So, we need to "fix" #3 to be one of those forms.
161  */
162  if (std::wstring::npos == wstr.find(L"x0 = "))
163  wstr.insert(0, L"x0 = 0\r\ndx = ");
164  wchar_t wch[max_len + 1] = { 0 }; // Make as large as str to be safe.
165  int count = 0;
166  try {
167  double d_x0 = 0, d_dx = 0;
168  count = swscanf_s(wstr.c_str(), L"x0 = %lf\r\ndx = %lf %[^\a]", &d_x0, &d_dx,
169  &wch, static_cast<unsigned int>(max_len + 1)); // \a is Bell. It'll never be in the string.
170  if (count >= 2) {
171  x0_ = d_x0;
172  dx_ = d_dx;
173  units_ = wch;
174  // Trim first and last character. Based on testing, they will be ( and ) respectively.
175  if (units_.length() >= 2)
176  units_ = units_.substr(1, units_.length() - 2);
177  return true;
178  }
179  return false;
180  }
181  catch (...) {
182  reset();
183  return false;
184  }
185  }
186  };
