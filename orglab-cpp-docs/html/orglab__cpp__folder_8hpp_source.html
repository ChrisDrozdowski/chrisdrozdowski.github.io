74  folder add_subfolder(std::wstring path) {
75  if (!ptr_)
76  throw std::exception(OLCP_EXCEPTION_MSG(msgs::class_ptr_invalid));
77 
78  // Default current folder.
79  orglab::FolderPtr folder_ptr = ptr_;
80 
81  // Clean up. Fix slashes, trim white space.
82  std::replace(path.begin(), path.end(), L'\\', L'/');
83  std::size_t first = path.find_first_not_of(L" \r\n\t\f\v");
84  if (std::wstring::npos != first) {
85  std::size_t last = path.find_last_not_of(L" \r\n\t\f\v");
86  path = path.substr(first, last - first + 1);
87  }
88  // If path is empty, skip adding.
89  if (path.length() > 0) {
90  // Trim slashes.
91  if (path.front() == '/')
92  path.erase(0, 1);
93  if (path.back() == '/')
94  path.pop_back();
95 
96  // Again, if path is empty, skip adding.
97  if (path.length() > 0) {
98  // Check if subfolder exists already.
99  folder_ptr = ptr_->Folders->FolderFromPath(_bstr_t(path.c_str()));
100  if (!folder_ptr) {
101  // No, then break path into parts.
102  std::vector<std::wstring> vec;
103  std::wstringstream ss(path);
104  std::wstring part;
105  while (getline(ss, part, L'/')) {
106  vec.push_back(part);
107  }
108 
109  // Start at current folder.
110  folder_ptr = ptr_;
111 
112  // Iterate parts heirarchically.
113  orglab::FolderPtr temp_ptr = nullptr;
114  for (const std::wstring& str : vec) {
115  // Does part exist as a subfolder? If not, add. If yes, move to next part.
116  temp_ptr = folder_ptr->Folders->FolderFromPath(_bstr_t(str.c_str()));
117  if (!temp_ptr)
118  folder_ptr = folder_ptr->Folders->Add(_variant_t(str.c_str()));
119  else
120  folder_ptr = temp_ptr;
121  }
122  }
123  }
124  }
125  return folder(folder_ptr);
126  }
