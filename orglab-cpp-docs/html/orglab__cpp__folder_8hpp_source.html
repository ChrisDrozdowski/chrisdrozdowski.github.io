131  folder add_subfolder(std::wstring path) {
132  if (!ptr_)
133  throw std::exception(OLCP_EXCEPTION_MSG(msgs::class_ptr_invalid));
134 
135  // Default current folder.
136  orglab::FolderPtr folder_ptr = ptr_;
137 
138  // Clean up. Fix slashes, trim white space.
139  std::replace(path.begin(), path.end(), L'\\', L'/');
140  std::size_t first = path.find_first_not_of(L" \r\n\t\f\v");
141  if (std::wstring::npos != first) {
142  std::size_t last = path.find_last_not_of(L" \r\n\t\f\v");
143  path = path.substr(first, last - first + 1);
144  }
145  // If path is empty, skip adding.
146  if (path.length() > 0) {
147  // Trim slashes.
148  if (path.front() == '/')
149  path.erase(0, 1);
150  if (path.back() == '/')
151  path.pop_back();
152 
153  // Again, if path is empty, skip adding.
154  if (path.length() > 0) {
155  // Check if subfolder exists already.
156  folder_ptr = ptr_->Folders->FolderFromPath(_bstr_t(path.c_str()));
157  if (!folder_ptr) {
158  // No, then break path into parts.
159  std::vector<std::wstring> vec;
160  std::wstringstream ss(path);
161  std::wstring part;
162  while (getline(ss, part, L'/')) {
163  vec.push_back(part);
164  }
165 
166  // Start at current folder.
167  folder_ptr = ptr_;
168 
169  // Iterate parts heirarchically.
170  orglab::FolderPtr temp_ptr = nullptr;
171  for (const std::wstring& str : vec) {
172  // Does part exist as a subfolder? If not, add. If yes, move to next part.
173  temp_ptr = folder_ptr->Folders->FolderFromPath(_bstr_t(str.c_str()));
174  if (!temp_ptr)
175  folder_ptr = folder_ptr->Folders->Add(_variant_t(str.c_str()));
176  else
177  folder_ptr = temp_ptr;
178  }
179  }
180  }
181  }
182  return folder(folder_ptr);
183  }
