280  void randomize_char_buffer(char* buf, std::size_t len) {
281  auto func = []() {
282  std::random_device rd;
283  std::array<int, std::mt19937::state_size> seed_data;
284  std::generate_n(seed_data.data(), seed_data.size(), std::ref(rd));
285  std::seed_seq seq(std::begin(seed_data), std::end(seed_data));
286  return std::mt19937(seq);

299  std::string unique_string(const char* pattern)
300  {
301  std::string str(pattern);
302  const char hex[] = "0123456789abcdef";
303  const char percent = '%';
304  char ran[] = "123456789abcdef";
305  const int max_nibbles = 2 * sizeof(ran); // 4-bits per nibble.
306  int nibbles_used = max_nibbles;
307  for (std::size_t i = 0; i < str.size(); ++i) {
308  if (str[i] == percent) {
309  if (nibbles_used == max_nibbles) {
310  randomize_char_buffer(ran, sizeof(ran));
311  nibbles_used = 0;
312  }
313  int c = ran[nibbles_used / 2];
314  c >>= 4 * (nibbles_used++ & 1); // If odd, shift right 1 nibble.
315  str[i] = hex[c & 0xf]; // Convert to hex digit and replace.
316  }
317  }
318  return str;
319  }
