23  inline com_compat_info_t get_com_compat_info(const orglab::COLDATAFORMAT& fmt) {
24  if (orglab::COLDATAFORMAT::DF_DATE == fmt || orglab::COLDATAFORMAT::DF_TIME == fmt) return com_compat_info_t{ fmt, VT_R8 };
25  if (std::is_same<T, double>::value) return com_compat_info_t{ orglab::COLDATAFORMAT::DF_DOUBLE, VT_R8 };
26  if (std::is_same<T, float>::value) return com_compat_info_t{ orglab::COLDATAFORMAT::DF_FLOAT, VT_R4 };
27  if (std::is_same<T, int>::value) return com_compat_info_t{ orglab::COLDATAFORMAT::DF_LONG, VT_I4 };
28  if (std::is_same<T, long>::value) return com_compat_info_t{ orglab::COLDATAFORMAT::DF_LONG, VT_I4 };
29  if (std::is_same<T, unsigned long>::value) return com_compat_info_t{ orglab::COLDATAFORMAT::DF_ULONG, VT_I4 };
30  if (std::is_same<T, short>::value) return com_compat_info_t{ orglab::COLDATAFORMAT::DF_SHORT, VT_I2 };
31  if (std::is_same<T, unsigned short>::value) return com_compat_info_t{ orglab::COLDATAFORMAT::DF_USHORT, VT_I2 };
32  if (std::is_same<T, std::wstring>::value) return com_compat_info_t{ orglab::COLDATAFORMAT::DF_TEXT_NUMERIC, VT_BSTR };
33  if (std::is_same<T, std::string>::value) return com_compat_info_t{ orglab::COLDATAFORMAT::DF_TEXT_NUMERIC, VT_BSTR };
34  if (std::is_same<T, byte>::value) return com_compat_info_t{ orglab::COLDATAFORMAT::DF_BYTE, VT_I1 };
35  if (std::is_same<T, char>::value) return com_compat_info_t{ orglab::COLDATAFORMAT::DF_CHAR, VT_I1 };
36  if (std::is_same<T, std::complex<double>>::value) return com_compat_info_t{ orglab::COLDATAFORMAT::DF_COMPLEX, VT_R8 };
37 
38  throw std::exception(OLCP_EXCEPTION_MSG(msgs::invalid_data_type));

219  if (std::is_same<T, int>::value && orglab::COLDATAFORMAT::DF_LONG == fmt) return true;

265  }

293  vec[i] = from_bstr(*p_csa);
294  }
295  ::SafeArrayUnlock(csa.m_psa);
296  return vec;

328  template<class T>
329  std::vector<std::complex<double>> get_complex_column_data(const orglab::ColumnPtr& col, const long& offset, const long& rows) {
330  std::vector<std::complex<double>> vec;
331  _variant_t vt_data = do_get_col_data(col, orglab::ARRAYDATAFORMAT::ARRAY1D_NUMERIC, to_non_negative_long(offset), rows < -1 ? -1 : rows);
332  if (VT_ARRAY & vt_data.vt) {
333  long lbound, ubound;
334  ::SafeArrayGetLBound(vt_data.parray, 1, &lbound);
335  ::SafeArrayGetUBound(vt_data.parray, 1, &ubound);
336  long count = ubound - lbound + 1;
337  if (count > 0) {
338  // OrgLab returns pattern below for complex data:
339  // c1.re,c1.im,c2.re,c2.im,c3.re,c3.im, etc.
340  // There will be twice as many values as is needed
341  // for a complex vector since complex has two parts.
342  // Use resize to already add default complex values
343  // to vector.
344  vec.resize(count / 2);
345  double* p_val = nullptr;
346  ::SafeArrayAccessData(vt_data.parray, (void**)&p_val);
347  for (std::size_t i = 0; i < vec.size(); ++i, ++p_val) {
348  vec[i].real(*p_val);
349  vec[i].imag(*(++p_val));
350  }
351  ::SafeArrayUnaccessData(vt_data.parray);
352  }
353  }
354  return vec;
355  }

392  template<class T> // Need T though we don't use for compilation.
393  matrix_adapter<std::complex<double>> get_complex_matrix_data(const orglab::MatrixObjectPtr& mat) {
394  matrix_adapter<std::complex<double>> ma;
395  CComSafeArray<double> csa;
396  { // Scope releases vt_data as soon as we hand its SAFEARRAY off to the CComSafeArray.
397  _variant_t vt_data = do_get_mat_data(mat, orglab::ARRAYDATAFORMAT::ARRAY2D_NUMERIC);
398  if (VT_ARRAY & vt_data.vt) {
399  csa.Attach(vt_data.parray);
400  vt_data.Detach();
401  }
402  }
403  // Despite ARRAY2D_NUMERIC, 3D array is returned for complex.
404  // CComSafeArray handles such arrays easier, that's why we use one. Little to no overhead.
405  mat_dim_t rows = to_mat_dim_t(csa.GetCount(0));
406  mat_dim_t cols = to_mat_dim_t(csa.GetCount(1));
407  mat_dim_t parts = to_mat_dim_t(csa.GetCount(2)); // Two parts- 1st is real part, 2nd is imaginary part.
408  // There can be no empty matrix object in Origin- always have at least 1x1.
409  if (rows > 0 && cols > 0 && parts > 0) {
410  ma.resize(rows, cols);
411  ::SafeArrayLock(csa.m_psa);
412  double* p_csa_real = (double*)(csa.m_psa->pvData); // Real part is first rows*cols values.
413  long offset = (long)(rows * cols); // Avoids possible arith overflow per VS code analysis.
414  double* p_csa_imag = p_csa_real + offset;// (rows * cols); // Imaginary part real part offset by rows*cols.
415  for (mat_dim_t c = 0; c < cols; ++c) {
416  for (mat_dim_t r = 0; r < rows; ++r, p_csa_real++, p_csa_imag++) {
417  ma(r, c) = std::complex<double>{ *p_csa_real , *p_csa_imag };
418  }
419  }
420  ::SafeArrayUnlock(csa.m_psa);
421  }
422  return ma;
423  }
