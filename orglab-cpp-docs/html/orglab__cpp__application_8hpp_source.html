16  // Custom function replaces default function used for COM error handling.
17  // Converts COM error to std::exception.
18  // Set via _set_com_error_handler(orglab_cpp_com_raise_error);
19  void __stdcall orglab_cpp_com_raise_error(HRESULT hr, IErrorInfo* perrorinfo) {
20  if (perrorinfo) {
21  BSTR bstr;
22  perrorinfo->GetDescription(&bstr);
23  char* ch = ::_com_util::ConvertBSTRToString(bstr);
24  ::SysFreeString(bstr);
25  std::string str(ch);
26  delete[] ch;
27  throw std::exception(str.c_str());
28  }
29  else
30  throw std::exception("Unknown COM Error");
31  }
32 
33  // Ensures that above function is called only one time.
34  void orglab_cpp_set_com_error_handler() {
35  static std::once_flag flag;
36  std::call_once(flag, []() {
37  ::_set_com_error_handler(orglab_cpp_com_raise_error);
38  });
39  }

170  return_values open_project(const std::wstring& file, bool read_only = false) {
171  file_ = L"";
172  name_ = default_file_name;
173  if (!app_)
174  throw std::exception(OLCP_EXCEPTION_MSG(msgs::ccomptr_invalid));
175  if (!is_file(file))
176  return return_values::file_exist_fail;
177  _variant_t v_file(file.c_str());
178  _variant_t v_readonly(read_only);
179  if (!app_->Load(v_file, v_readonly)) {
180  std::error_code ec;
181  if (std::filesystem::file_size(file, ec) < 5)
182  return return_values::invalid_file;
183 
184  FILE* f;
185  //errno_t err = fopen_s(&f, from_wide(file).c_str(), "rb");
186  errno_t err = _wfopen_s(&f, file.c_str(), L"rb");
187  if (0 != err || !f)
188  return return_values::invalid_file;
189 
190  char buf[3] = { 0 };
191  std::size_t sz = fread(&buf[0], sizeof(char), 3, f);
192  fclose(f);
193  if (3 != sz)
194  return return_values::invalid_file;
195 
196  if ((buf[0] != 'C') || (buf[1] != 'P') || (buf[2] != 'Y'))
197  return return_values::invalid_file;
198 
199  return return_values::open_fail;
200  }
201  file_ = file;
202  name_ = std::filesystem::path(file).filename();
203  return return_values::ok;
204  }
